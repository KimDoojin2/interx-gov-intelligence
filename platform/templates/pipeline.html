Add this code at the end of your Colab pipeline to auto-sync results to this platform: 

import requests, json

PLATFORM_URL = "http://YOUR_SERVER:8000"  # Replace with your platform URL

def sync_to_platform(result, notices_data):
    """Send pipeline results to BD Platform."""
    payload = {
        "execution_id": result.get("execution_id", ""),
        "notice_count": result.get("notice_count", 0),
        "l3_count": len(result.get("l3_rows", [])),
        "a_count": sum(1 for n in notices_data if n.get("grade") == "A"),
        "b_count": sum(1 for n in notices_data if n.get("grade") == "B"),
        "error_count": result.get("error_count", 0),
        "notices": notices_data,
    }
    resp = requests.post(f"{PLATFORM_URL}/api/pipeline/sync", json=payload)
    print(f"Platform sync: {resp.json()}")
